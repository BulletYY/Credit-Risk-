from sklearn.model_selection import StratifiedKFold
from sklearn.linear_model import LogisticRegression
from sklearn.metrics import accuracy_score

wyniki_accuracy_cv_train = []

wyniki_accuracy_cv_val = []

cv = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)

kolumny = ["A3", "A6", "A7", "A8", "A9"]

for d in range(1000):
    probkowane = pd.concat([
        
        wybrane_do_mc[wybrane_do_mc["true_variable"] == 1].sample(100, replace=True),
        wybrane_do_mc[wybrane_do_mc["true_variable"] == 0].sample(100, replace=True)
    ])  

    probkowane = probkowane.dropna(subset=kolumny + ["true_variable"]) # usuwanie nan

    X = probkowane[kolumny]
    y = probkowane["true_variable"]

    fold_train_scores = []
    fold_val_scores = []

    for train_idx, val_idx in cv.split(X, y):

        X_train = probkowane.iloc[train_idx][kolumny]
        y_train = probkowane.iloc[train_idx]["true_variable"]

        X_val = probkowane.iloc[val_idx][kolumny]
        y_val = probkowane.iloc[val_idx]["true_variable"]

        model = LogisticRegression(max_iter=1000)
        model.fit(X_train, y_train)

        fold_train_scores.append(accuracy_score(y_train, model.predict(X_train))) # fit na treningowe dane
        fold_val_scores.append(accuracy_score(y_val, model.predict(X_val))) # predykcja na te niewidziane wczesniej (walidacja)

    wyniki_accuracy_cv_train.append(np.mean(fold_train_scores))
    wyniki_accuracy_cv_val.append(np.mean(fold_val_scores))
